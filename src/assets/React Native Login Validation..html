    import { createSlice, PayloadAction } from '@reduxjs/toolkit';

interface Expense {
  id: string;
  title: string;
  amount: number;
  date: Date;
}

interface ExpensesState {
  expensesByDate: Record<string, Expense[]>;
}

const initialState: ExpensesState = {
  expensesByDate: {},
};

const expensesSlice = createSlice({
  name: 'expenses',
  initialState,
  reducers: {
    addExpense: (state, action: PayloadAction<Expense>) => {
      const { date } = action.payload;
      const formattedDate = date.toDateString();

      if (!state.expensesByDate[formattedDate]) {
        state.expensesByDate[formattedDate] = [];
      }

      state.expensesByDate[formattedDate].push(action.payload);
    },
    deleteExpense: (state, action: PayloadAction<string>) => {
      const expenseId = action.payload;

      Object.keys(state.expensesByDate).forEach((date) => {
        const expenses = state.expensesByDate[date];
        const index = expenses.findIndex((expense) => expense.id === expenseId);

        if (index !== -1) {
          expenses.splice(index, 1);
          if (expenses.length === 0) {
            delete state.expensesByDate[date];
          }
          return;
        }
      });
    },
  },
});

export const { addExpense, deleteExpense } = expensesSlice.actions;

export default expensesSlice.reducer;
